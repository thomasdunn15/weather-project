# Clean up Herbie GRIB caches every hour — purge anything older than 60 min.
# Hourly cadence (vs old daily) and short retention (vs old +3 days) are
# required because backfills can churn through >50 GB/day per worker, faster
# than daily cleanup can keep up. Disk-full once crashed Postgres and locked
# us out of SSH on 2026-05-30, so this is safety-critical. Herbie re-downloads
# on cache miss, so aggressive cleanup is harmless to correctness.
0 * * * * find /home/tdunn/data/gefs -type f -mmin +180 -delete
0 * * * * find /home/tdunn/data/hrrr -type f -mmin +180 -delete

# ECMWF 00 UTC run (all registered stations): try at 07:00 (normal publication
# ~04:00 UTC + buffer), retry at 13:00. The 07:00 run catches the 90% case;
# the 13:00 retry catches days when Open Data is delayed (which has happened).
# Both runs are idempotent — the second is a no-op if the first succeeded, and
# rescues today's IFS when the first fell back to yesterday. 13:00 still leaves
# 1h 45min before the 14:45 paper-trade cron.
0 7 * * * cd /home/tdunn/weather-project && /home/tdunn/.local/bin/uv run python scripts/ingest_ecmwf_daily.py --run-hour 0 >> /var/log/weather/ecmwf_00z.log 2>&1
0 13 * * * cd /home/tdunn/weather-project && /home/tdunn/.local/bin/uv run python scripts/ingest_ecmwf_daily.py --run-hour 0 >> /var/log/weather/ecmwf_00z.log 2>&1

# ECMWF 12 UTC run (all registered stations): try today (open data publishes ~17 UTC), fall back to yesterday on failure
30 18 * * * cd /home/tdunn/weather-project && /home/tdunn/.local/bin/uv run python scripts/ingest_ecmwf_daily.py >> /var/log/weather/ecmwf.log 2>&1

# HRRR 00Z run (all registered stations): try at 03:30 UTC (HRRR open data typically lands ~02-03 UTC).
# Retry at 13:45 UTC — 1 hour before KORD live cron at 14:46 — to catch days where AWS publication is

# REAL-TIME fill monitor: 15-second polling for near-instant dashboard updates.
# Launches a tmux session at 14:45 UTC (right when KORD's live cron places
# orders) that runs monitor_fills.py --loop 15 --until 19:55. Polls Kalshi
# every 15s, updates DB as partial fills come in. Auto-terminates at 19:55
# UTC (right before the 20:00 EOD cancel-unfilled cron). Idempotent: if a
# session already exists, the new-session command is a no-op.
